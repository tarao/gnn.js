 55     /**
 56         Creates a hash table.
 57         @class A hash table.
 58         @name H
 59         @exports H as GNN.Hash
 60         @param {GNN.AssocArray|object|...*} []
 61             If the odd number of arguments are specified,
 62             then the first argument must be a GNN.AssocArray or an object.
 63             The keys and values of the argument are copied to the hash table.
 64             Otherwise, the arguments must be an array.
 65             The <code>i</code>th element of the array is a key
 66             and the <code>(i+1)</code>th element of the array is the
 67             value associated with the <code>i</code>th key
 68             for every even number <code>i</code>.
 69         @description
 70             <p>Methods and property gets in <code>GNN.Hash.prototype</code>
 71             are also available as static methods in <code>GNN.Hash</code> with
 72             taking the first argument as an array. In this form, an object
 73             can be used as a hash.</p>
 74         @example
 75 new GNN.Hash({a:1, b:2, c:3});
 76 // => Hash {a: 1, b:2, c:3}
 77 new GNN.Hash(new GNN.AssocArray(['a', 1], ['b', 2], ['c': 3]));
 78 // => Hash {a: 1, b:2, c:3}
 79 new GNN.Hash('a', 1, 'b', 2, 'c', 3);
 80 // => Hash {a: 1, b:2, c:3}
 81     */
 82     H = T.Hash = function Hash(hashLike) {
 83         var self = this;
 84         if (!(self instanceof H)) self = new H();
 85 
 86         if (arguments.length > 0 && arguments.length % 2 == 0) {
 87             hashLike = toA(arguments);
 88         }
 89         if (T.AssocArray && T.AssocArray.isAssocArray(hashLike)) {
 90             hashLike.forEach(function(x){ self[x[0]] = x[1]; });
 91         } else if (hashLike instanceof Array && hashLike.length % 2 == 0) {
 92             for (var i=0; i < hashLike.length; i+=2) {
 93                 self[hashLike[i]] = hashLike[i+1];
 94             }
 95         } else {
 96             // clone
 97             fmerge(function(a, b, k){ self[k] = b; }, null, hashLike);
 98         }
 99 
100         return self;
101     };
