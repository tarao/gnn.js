227     var deep = {
228         Label: function() { return {
229             labels: [], seen: {},
230             add: function(l) { if (!this.seen[l]) {
231                 this.labels.push(l); this.seen[l] = true;
232             } } }; },
233         Seen: function(seen) { var K; return new (K=function(seen) { return {
234             clone: function(){ return new K(seen.slice(0)); },
235             add: function(l, r) {
236                 for (var i=0; i < seen.length; i++) {
237                     if (seen[i][0] === l && seen[i][1] === r) return i+1;
238                 }
239                 seen.push([ l, r ]); } }; })(seen||[]); },
240         refl: function(lhs, rhs) {
241             if ((typeof lhs) != (typeof rhs)) {
242                 return false;
243             } else if (lhs == null && rhs == null) {
244                 return true;
245             } else if (lhs == null || rhs == null) {
246                 return false;
247             } else if (lhs === rhs) {
248                 return true;
249             } else if (B.isObject(lhs) && B.isObject(rhs)) {
250                 return true;
251             }
252             return false;
253         },
254         indexes: function(a, b, l) {
255             var length = 0;
256             if (B.isArray(a)) length = Math.max(a.length, length);
257             if (B.isArray(b)) length = Math.max(b.length, length);
258             for (var i=0; i < length; i++) {
259                 if ((i in a) || (i in b)) l.add(i);
260             }
261         },
262         keys: function(obj, l){ for (var k in obj) l.add(k); },
263         labels: function(lhs, rhs, l) {
264             this.indexes(lhs, rhs, l); this.keys(lhs, l); this.keys(rhs, l);
265         },
266         labelsRight: function(lhs, rhs, l) {
267             this.indexes(rhs, rhs, l); this.keys(rhs, l);
268         },

270             if (!(B.isObject(lhs) && B.isObject(rhs))) return []; // leaf
271             var l = new this.Label(); labels.call(this, lhs, rhs, l);
272 
273             var r = [];

275                 r.push([ lhs[l.labels[i]], rhs[l.labels[i]] ]);

281 
282             // detect cyclic reference
283             if ((seen=seen.clone()).add(lhs, rhs)) return true;

285             var states = transition.call(this, lhs, rhs);
286             for (var i=0; i < states.length; i++) {
287                 var l = states[i][0]; var r = states[i][1];
288                 if (!this.sim(l, r, seen, axiom, transition)) return false;

294     /**
295         Checks if there is a simulation over the given
296         labelled state transition system.
297         @param {*} lhs
298         @param {*} rhs
299         @param {function} axiom
300             A function which takes two states and returns a boolean value.
301         @param {function} transition
302             A function which takes two states and returns
303             an array of pairs of states.
304         @returns {boolean}
305         @description
306             <p>The labelled state transition is given by values as states and
307             indexes of a return value of <code>transition</code> as labels.
308             The return value of <code>transition</code> is an array of pairs
309             of next states.</p>
310 
311             <p>The overall result is <code>true</code> if and only if
312             <code>axiom(lhs, rhs)</code> returns <code>true</code> and
313             <code>sim(x[i], y[i], axiom, transition)</code> returns
314             <code>true</code> for every <code>i</code> where
315             <code>transition(lhs, rhs)</code> returns
316             <code>[ [ x[0], y[0] ], ..., [ x[i], y[i] ], ... ]</code>.</p>
317 
318             <p>For any states <code>x1</code> and <code>y1</code>,
319             <code>sim(x1, y1, axiom, transition) = true</code> implies
320             <code>sim(x2, y2, axiom, transition) = true</code> for every
321             pair of states <code>x2</code>, <code>y2</code> returned by
322             <code>transition(x1, y1)</code>.
323             This means that there is a relation called
324             <a href="http://en.wikipedia.org/wiki/Simulation_preorder">simulation</a>
325             over the transition system
326             and the relation is actually a
327             <a href="http://en.wikipedia.org/wiki/Bisimulation">bisimulation</a>
328             if <code>axiom</code> is
329             symmetric.</p>
330 
331             <p>This method is useful for checking observable equivalence.</p>
332         @see GNN.Base.eq
333         @see GNN.Base.covers
334     */
335     B.sim = function(lhs, rhs, axiom, transition) {
336         axiom = axiom || deep.refl;
337         transition = transition || function(lhs, rhs) {
338             return deep.properties(deep.labels, lhs, rhs);
339         };
340         return deep.sim(lhs, rhs, new deep.Seen(), axiom, transition);
341     };

399     /**
400         Makes a visitor for structured objects.
401         @class A visitor for structured objects.
402         @param {function} kindOf
403             A function which takes a value and returns a kind of the value in
404             a string.
405         @param {object} methods
406             A map from kinds of values to visitor functions for each kind.
407         @see GNN.Base.pp
408         @example
409 var visitor = new GNN.Base.Visitor(function(obj) {
410     return typeof obj;
411 }, {
412     undefined: function(){ return 0; },
413     null: function(){ return 0; },
414     number: function(){ return 1; },
415     string: function(){ return 1; },
416     function: function(){ return 1; },
417     object: function(obj) {
418         var n = 0;
419         for (var x in obj) {
420             n += this.visit(obj[x]);
421         }
422         return n;
423     }
424 });
425 visitor.visit({ a: 1, b: [ 1, 2, { c: 3, d: 4 } ], e: 5 }); // => 6
426     */
427     B.Visitor = function(kindOf, methods) {
428         var visitor = function(seen) { return {
429             /** @lends B.Visitor.prototype */
430             /**
431                 Determines the kind of the given value and dispatches to
432                 the visitor methods.
433                 @param {*} obj
434                 @returns The return value of the visitor method.
435                 @description
436                     This method can be called recursively from inside
437                     the visitor methods via <code>this.visit(obj)</code>.
438                     It dispatches to 'cyclic' visitor method
439                     if the cyclic reference is found.
440             */
441             visit: function(obj) {
442                 var args = [ obj ];
443                 var s = seen.clone(); var i = s.add(obj, obj);
444                 var k = kindOf(obj);
445                 if (i) {
446                     k = 'cyclic';
447                     args.push(i);
448                 }
449                 if (B.isCallable(methods[k])) {
450                     return methods[k].apply(visitor(s), args);
451                 }
452             }
453         }; };
454         return visitor(new deep.Seen());
455     };

457     var escape = function(str, q) {
458         var bs = new RegExp('\\\\', 'g'); q = q || '"';
459         str = str.replace(bs, '\\\\').replace(new RegExp(q, 'g'), '\\'+q);

465     var ppFun = function(fun, args) {
466         var str = fun+'';
467         if (/^\[object (.*)\]$/.test(str)) {

507     /**
508         Pretty prints the given value.
509         @param {*} obj
510             A value to be printed.
511         @param {object} [args]
512             Options in a hash table.
513         @param {boolean} [args.prettify=false]
514               Shows function bodies indented (if possible) and uses
515               <code>Name({prop:val,...})</code> notation for objects
516               together with <code>args.object.name=true</code>, and
517               <code>cyclic(i)</code> notation for cyclic references.
518         @param {boolean} [args.function.detail=true]
519               Shows function bodies.
520         @param {boolean} [args.object.name=true]
521               Uses <code>Name {prop:val,...}</code> notation.
522         @param {boolean} [args.cyclic.detail=false]
523               Uses <code>&i</code> notation.
524         @returns {string}
525         @requires Firefox (to get function bodies to be indented with
526                   <code>args.prettify=true</code>.)
527         @see GNN.Base.prettify
528         @example
529 var a1 = [1,2,3];
530 GNN.Base.pp(a1); // => '[1, 2, 3]'
531 
532 var a2 = [1,2,3];
533 a2[1] = a2;
534 GNN.Base.pp(a2); // => '[1, ..., 3]'
535 GNN.Base.pp(a2,{cyclic:{detail:1}}); // => '[1, &1, 3]'
536 GNN.Base.pp(a2,{prettify:1}); // => '[1, cyclic(1), 3]'
537 
538 var x = { a:1, b:2, c:3 };
539 GNN.Base.pp(x); // => '{a: 1, b: 2, c: 3}'
540 
541 var SomeClass = function SomeClass(a,b){ this.a=a; this.b=b; };
542 var y = new SomeClass(1,2);
543 GNN.Base.pp(y); // => 'SomeClass {a: 1, b: 2}'
544 GNN.Base.pp(y,{prettify:1,object:{name:false}}); // => '{a: 1, b: 2}'
545 GNN.Base.pp(y,{object:{name:false}}); // => '{a: 1, b: 2}'
546 GNN.Base.pp(y,{prettify:1}); // => 'SomeClass({a: 1, b: 2})'
547     */
548     B.pp = function(obj, args) {
549         args = args || {};
550         var fargs = args['function'] || { indent: args.prettify };
551         var fdetail = !('detail' in fargs) || fargs.detail;
552         var prettify = args.prettify && fdetail;
553 
554         var objName = function(klass, x){ return x; };
555         if (!('name' in (args.object||{})) || (args.object||{}).name) {
556             if (prettify) {
557                 /** @ignore */
558                 objName = function(klass, x){ return klass+'('+x+')'; };
559             } else {
560                 /** @ignore */
561                 objName = function(klass, x){ return klass+' '+x; };
562             }
563         }
564 
565         var str = new B.Visitor(B.kindOf, {
566             atom: function(atom) {
567                 return (typeof atom == 'string') ? ppStr(atom) : atom+'';
568             },
569             function: function(fun){ return ppFun(fun, fargs); },
570             array: function(arr) {
571                 var ss = [];
572                 for (var i=0; i < arr.length; i++) ss.push(this.visit(arr[i]));
573                 return '['+ss.join(', ')+']';
574             },
575             object: function(obj) {
576                 var name = function(klass, x){ return x; };
577                 var klass = B.className(obj);
578                 if (klass != 'Object') name = objName;
579 
580                 if (B.isObject(obj) && !B.isECMAObject(obj)) {
581                     if (B.isDefined(document) && obj instanceof Element) {
582                         try {
583                             var div = document.createElement('div');
584                             div.appendChild(obj);
585                             return div.innerHTML;
586                         } catch (e) { /* ignore */ }
587                     }
588                     // if (B.className(obj) != 'Object') {
589                         var str = '{...}';
590                         if (prettify) str = 'omitted';
591                         return name(klass, str);
592                     // }
593                 }
594 
595                 var ss = [];
596                 for (var k in obj) {
597                     var val = this.visit(obj[k]);
598                     if (!/^[a-zA-Z$_][a-zA-Z0-9$_]*$/.test(k)) {
599                         k = "'"+escape(k, "'")+"'";
600                     }
601                     ss.push(k + ': ' + val);
602                 }
603                 return name(klass, '{'+ss.join(', ')+'}');
604             },
605             cyclic: function(obj, i) {
606                 if (prettify) return 'cyclic('+i+')';
607                 if ((args.cyclic||{}).detail) return '&'+i;
608                 return '...';
609             }
610         }).visit(obj);
611 
612         if (prettify) {
613             return B.prettify('return '+str).replace(/^return /,'');
614         }
615         return str;
616     };

621     /**
622         Merges the objects by the given function.
623         @param {function} fun
624             A function takes two values and returns a value or undefined.
625         @param {...object} objs
626             Zero or more objects to be merged.
627         @returns {object}
628             The value whose reference is identical to the first element of
629             <code>objs</code>.
630         @description
631             <p>For input <code>obj1, obj2, ...rest</code>,
632             it merges <code>obj1</code> and <code>obj2</code>
633             into <code>obj1</code> and computes
634             <code>fmerge(fun, obj1, ...rest)</code> iteratively.
635             In each iteration, it calls <code>fun(obj1[p], obj2[p], p)</code>
